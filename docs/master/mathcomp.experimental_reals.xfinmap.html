    Lemma big_nat_mkfset (F : nat -> R) n :
      \sum_(0 <= i < n) F i =
        \sum_(i : [fset x in (iota 0 n)]) F (val i).
    Proof.
    rewrite -(big_map val xpredT) /=; apply/perm_big.
    apply/uniq_perm; rewrite ?iota_uniq //.

    Qed.
